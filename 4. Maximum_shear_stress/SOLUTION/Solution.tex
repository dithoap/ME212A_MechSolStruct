We assume that during a tensile test (Fig. \ref{fig:uniaxial}), the stress state is homogeneous within the part $\Omega$ of a material sample. For any material point of $\Omega$, the local stress tensor is:


\begin{equation}
\sigmauu = 100 \cdot \eu_1 \otimes \eu_1 
\end{equation}

\begin{description}
	\item{\textbf{Question 1}} Calculate the global force $\Vec{F}$ that is applied on the material sample by the experimental tensile device.
	\item{\textbf{Question 2}} We cut the sample by a ``fictive" plane. The orientation of this plane is denoted by the angle $\theta$ between the $\Vec{e}_2$ axis and the unit normal vector $\Vec{n}$. Calculate as a function of $\theta$ the stress vector applied on this surface.
	\item{\textbf{Question 3}} Find the value of $\theta$ for which the shear stress is maximum. 
\end{description}

\newpage
\section*{Solution to "Maximum shear stress"} 


	\noindent \textbf{Question 1} Calculate the global force $\Vec{F}$ that is applied on the material sample by the experimental tensile device. \\
	
	\textbf{Answer:} By definition the elementary force applied on the surface is
	\begin{equation}
	d\Fu = (\sigmauu \cdot \T{n} )dS
	\end{equation}
	where
	\begin{equation}
	\Fu = \int_S \sigmauu \cdot \T{n} dS = \int_S \sigma_{11} \eu_1 dS = S\sigma_{11} \eu_1
	\end{equation}

	Since $S = 10$ mm$^2$ and $\sigma_{11} = 100$ MPa, then $\Fu = 1000 $ (Newton) $ \eu_1$  \\
	
	
	\noindent \textbf{Question 2} We cut the sample by a ``fictive" plane. The orientation of this plane is denoted by the angle $\theta$ between the $\Vec{e}_2$ axis and the unit normal vector $\Vec{n}$. Calculate as a function of $\theta$ the stress vector applied on this surface. \\
	
	The normal direction of the fictive plane can be related as follows:
	\begin{equation}
	\T{n} = \sin \theta \eu_1 + \cos \theta \eu_2
	\end{equation}
	The stress vector at point M along the fictive plane is given by
	\begin{equation}
	\sigmauu \cdot \T{n} = \sigma_{11} \eu_1 \otimes \eu_1 \cdot (\sin \theta \eu_1 + \cos \theta \eu_2)
	\end{equation}
	\begin{equation}
	\sigmauu \cdot \T{n} = \sigma_{11} \sin \theta \eu_1
	\end{equation}
	If we transform the global orientation to the local orientation according to the fictive plane, then
	\begin{equation}
	\eu_1 = \sin \theta \T{n} + \cos\theta \tu
	\end{equation}
	and therefore,
	\begin{equation}
	\sigmauu \cdot \T{n} = \sigma_{11} \sin^2 \theta \T{n} + \sigma_{11} \sin \theta \cos \theta \tu
	\end{equation}
	where the first term is the normal component of the stress vector and the second term is the shear component  of the stress vector. \\
	
	\noindent \textbf{Question 3} Find the value of $\theta$ for which the shear stress is maximum.
	\begin{equation}
	\derp{(\sinteta\costeta)}{\theta} = \cos^2\theta - \sin^2\theta = 0 \rightarrow \theta = 45^\circ
	\end{equation}
